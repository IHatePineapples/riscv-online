229  static auto tokenize = [&data](const std::string &s) {
230  static std::size_t cursor = 0;
231  if (s.substr(0, 4) == ".equ")
232  cursor += 4;
233 
234  while (s[cursor] == ' ' || s[cursor] == ',')
235  ++cursor;
236 
237  auto symbol_start = cursor;
238  while (s[cursor] == ',')
239  ++cursor;
240  auto symbol_size = cursor - symbol_start;
241 
242  while (s[cursor] == ' ')
243  ++cursor;
244 
245  data[s.substr(symbol_start, symbol_size)] =
246  std::stoi(s.substr(cursor, s.length() - cursor));
247  };
248 
249  std::for_each(start + 1, last, tokenize);
250 
251  return data;

254 static std::optional<I> make_I(const std::string_view s) {
255  static std::size_t cursor = 0;
256  while (isalpha(s[cursor]))
257  ++cursor;
258  static auto opc = to_opc(s.substr(0, cursor));
259  while (!isalnum(s.at(cursor)))
260  ++cursor;
261  static auto opa1_i = cursor;
262  while (isalnum(s.at(cursor)))
263  ++cursor;
264  static auto opa1 = to_reg(s.substr(opa1_i, cursor - opa1_i));
265  while (!isalnum(s.at(cursor)))
266  ++cursor;
267  static auto opa2_i = cursor;
268  while (isalnum(s.at(cursor)))
269  ++cursor;
270  static auto opa2 = to_reg(s.substr(opa2_i, cursor - opa2_i));
271  while (!isalnum(s.at(cursor)))
272  ++cursor;
273 
274  static int imm = std::stoi(std::string(s.substr(cursor, s.length() - cursor)));
275  return std::make_tuple(opc, opa1, opa2, imm);
