29 inline std::string base36_encode(long long num)
30 {
31 int sign = num < 0;
32 std::string buff;
33
34 do
35 {
36 buff += digits.at(abs(num % 36));
37 num /= 36;
38 } while (num);
39
40 if (sign)
41 buff += '-';
42 std::reverse(buff.begin(), buff.end());
43 return buff;
44 }

56 template <std::size_t xlen>
57 inline std::vector<bool> decode_ram(const std::string_view s)
58 {
59 std::vector<bool> ram;
60
61 for (const auto l : std::views::split(s, delim))
62 {
63 uint32_t to_int = 0;
64 std::from_chars(l.begin(), l.begin() + l.size(), to_int, 36);
65 std::bitset<xlen> bs = to_int;
66
67 if (bs.any())
68 bs.set(0);
69
70 for (int i = xlen - 1; i != -1; --i)
71 ram.emplace_back(bs.test(i));
72 }
73
74 return ram;
75 }
